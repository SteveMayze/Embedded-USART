Non-blocking delay in ms.

#include "common.h"
#include "MCU/led.h"
#include "MCU/tick.h"
void main(void) {
TickType Delay;
Delay.DelayMs = 1000; //set to 1s
Led_Init();
Tick_init();
// reset the counter
Tick_DelayMs_NonBlocking(TRUE, &Delay);
for( ;;) {
if(Tick_DelayMs_NonBlocking(TRUE, &Delay)) {
// Delay has been reached
Tick_DelayMs_NonBlocking(TRUE, &Delay);
Led_Toggle();
}
else {
// User code when the code delay hasn't passed
}
}
}
